      # hw2
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

df = pd.read_csv("~/Desktop/CorporateFinance404/hw2/sp500.csv")
df = df.replace('C',0)
dividend_date = df.dropna().index.values
non_dividend = list()
return_mean = list()

for i in range(-20,21):
    x = [j + i for j in dividend_date]
    non_dividend.append(x)


for i in non_dividend:
    temp = df.iloc[i,3].astype("float").subtract(df.iloc[i,4])
    return_mean.append(np.mean(temp))

plt.plot([i for i in range(-20,21)],return_mean, 'r', label = 'My HW2')
plt.ylabel("Excess Return")
plt.xlabel("Days related to dividend date")

# HW3

df3 = pd.read_csv("~/Desktop/CorporateFinance404/hw5/hw3_huanyu.csv")
df3 = df3.replace('C',0)
df3 = df3.dropna()
df3['date'] = pd.to_datetime(df3['date'],format='%Y%m%d')
permno = df3['PERMNO'].unique()
no_comp = len(permno)
returns = np.zeros(no_comp)
market_cap = np.zeros(no_comp)

for i in range(no_comp):
    daily_return = df3[df3['PERMNO'] == permno[i]]
    sum = 0
    count = 0
    for j in range(2007,2018):
        annual = daily_return[(daily_return['date'] < str(j+1) + '-01-01') & (daily_return['date'] > str(j) + '-01-01')]
        if len(annual) > 0:
            count += 1
            annual_return = np.prod(annual['RET'].astype('float') + 1) - 1
            sum += annual_return
    returns[i] = sum / count


for i in range(no_comp):
    market_cap[i] = np.mean(np.absolute(df3[df3['PERMNO'] == permno[i]]['PRC'].astype('float')) * df3[df3['PERMNO'] == permno[i]]['SHROUT'].astype('float'))

summary = pd.DataFrame({'average_return':returns,'average marketcap': market_cap})
summary = summary.set_index(permno)

plt.plot(summary['average marketcap'],summary['average_return'],'ro')
plt.xlabel('average marketcap')
plt.ylabel('average annual return')
plt.title('Average return to average marketcap')
plt.show()
